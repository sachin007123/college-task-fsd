04 — Modules
// math.js
export const add = (a, b) => a + b;
export default class Calc {
  pow(a, b) { return a ** b; }
}

// app.js
import Calc, { add } from "./math.js";
const c = new Calc();
console.log(add(3,4), c.pow(2,8));
05 — Promises
const fetchData = id => new Promise((res, rej) =>
  setTimeout(() => id > 0 ? res("User: Alice") : rej("Not found"), 100)
);

fetchData(1)
  .then(data => console.log("✓", data))
  .catch(err => console.log("✗", err));
06 — Async / Await
const wait = (ms, v) => new Promise(r => setTimeout(() => r(v), ms));

async function main() {
  const user  = await wait(80, "Alice");
  const score = await wait(80, 95);
  console.log(`${user} scored ${score}`);
}

main();
